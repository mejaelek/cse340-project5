INSERT  Insert a new account record for Tony Stark. The account_id and account_type fields handle themselves (SERIAL + DEFAULT).

assignment2.sql — Query 1

-- account_id (SERIAL) and account_type (DEFAULT 'Client') are auto-handled

  account_firstname,

  account_email,
  account_password
)

  'Tony',

UPDATE  Modify Tony Stark's account type to 'Admin'. Uses email in the WHERE clause to target the exact record without accidentally changing others.

assignment2.sql — Query 2 
-- Query 2: Change Tony Stark's account_type to 'Admin'
UPDATE account
SET account_type = 'Admin'
WHERE account_email = 'tony@starkent.com';

DELETE  Remove Tony Stark's record permanently. Uses the primary key email to ensure only one row is affected.

⚠ Danger: DELETE without WHERE removes ALL rows. Always verify your WHERE clause.
assignment2.sql — Query 3 
-- Query 3: Delete the Tony Stark record from the database
DELETE FROM account
WHERE account_email = 'tony@starkent.com';

UPDATE  Use PostgreSQL's REPLACE() function to swap "small interiors" for "a huge interior" in the GM Hummer description — without retyping the full description.

assignment2.sql — Query 4 
-- Query 4: Update GM Hummer description using REPLACE()
-- DO NOT retype the entire description — use REPLACE()
UPDATE inventory
SET inv_description = REPLACE(
  inv_description,
  'small interiors',
  'a huge interior'
)
WHERE inv_make  = 'GM'
  AND inv_model = 'Hummer';

SELECT  Use an INNER JOIN to pull make, model, and classification name for all Sport vehicles. Joins inventory to classification on classification_id. Returns 2+ records.

assignment2.sql — Query 5 
-- Query 5: INNER JOIN inventory + classification
-- Returns make, model, and classification name for Sport vehicles
SELECT
  inv.inv_make,
  inv.inv_model,
  cls.classification_name
FROM   inventory       AS inv
INNER JOIN classification AS cls
  ON inv.classification_id = cls.classification_id
WHERE  cls.classification_name = 'Sport';

UPDATE  Use REPLACE() on both inv_image and inv_thumbnail in a single query to add /vehicles to every image path. After update paths should resemble /images/vehicles/car-name.jpg

assignment2.sql — Query 6 
-- Query 6: Update ALL inventory image paths using REPLACE()
-- Before: /images/car-name.jpg
-- After:  /images/vehicles/car-name.jpg
UPDATE inventory
SET
  inv_image     = REPLACE(inv_image,     '/images/', '/images/vehicles/'),
  inv_thumbnail = REPLACE(inv_thumbnail, '/images/', '/images/vehicles/');

Task 2 Database Rebuild — database-rebuild.sql
Important: The rebuild file is one file you run to recreate everything. It drops tables in the correct order, creates the ENUM type, creates all 3 tables, inserts data, then re-applies queries 4 and 6 from Task 1.
REBUILD database/database-rebuild.sql — Annotated Key Sections 
-- Step 1: DROP existing objects safely
DROP TABLE IF EXISTS inventory       CASCADE;
DROP TABLE IF EXISTS classification   CASCADE;
DROP TABLE IF EXISTS account         CASCADE;
DROP TYPE  IF EXISTS account_type    CASCADE;

-- Step 2: CREATE the ENUM type
CREATE TYPE account_type AS ENUM ('Client', 'Employee', 'Admin');

-- Step 3: CREATE classification table
CREATE TABLE classification (
  classification_id   SERIAL PRIMARY KEY,
  classification_name VARCHAR(30) NOT NULL UNIQUE
);

-- Step 4: CREATE account table
CREATE TABLE account (
  account_id        SERIAL PRIMARY KEY,
  account_firstname VARCHAR(50)  NOT NULL,
  account_lastname  VARCHAR(50)  NOT NULL,
  account_email     VARCHAR(100) NOT NULL UNIQUE,
  account_password  VARCHAR(200) NOT NULL,
  account_type      account_type NOT NULL DEFAULT 'Client'
);

-- Step 5: CREATE inventory table with FK to classification
CREATE TABLE inventory (
  inv_id            SERIAL PRIMARY KEY,
  inv_make          VARCHAR(50)   NOT NULL,
  inv_model         VARCHAR(50)   NOT NULL,
  inv_year          NUMERIC(4,0)  NOT NULL,
  inv_description   TEXT          NOT NULL,
  inv_image         VARCHAR(200)  NOT NULL,
  inv_thumbnail     VARCHAR(200)  NOT NULL,
  inv_price         NUMERIC(9,0)  NOT NULL,
  inv_miles         NUMERIC(9,0)  NOT NULL,
  inv_color         VARCHAR(50)   NOT NULL,
  classification_id INT           NOT NULL,
  CONSTRAINT fk_classification
    FOREIGN KEY (classification_id)
    REFERENCES classification(classification_id)
    ON DELETE CASCADE ON UPDATE CASCADE
);

-- Step 6: INSERT sample data (5 classifications, 3 accounts, 11 vehicles)
INSERT INTO classification (classification_name) VALUES
  ('Custom'), ('Sport'), ('SUV'), ('Truck'), ('Sedan');

-- Step 7: Re-apply Task 1 transformations
UPDATE inventory
SET inv_description = REPLACE(inv_description, 'small interiors', 'a huge interior')
WHERE inv_make = 'GM' AND inv_model = 'Hummer';

UPDATE inventory
SET
  inv_image     = REPLACE(inv_image,     '/images/', '/images/vehicles/'),
  inv_thumbnail = REPLACE(inv_thumbnail, '/images/', '/images/vehicles/');
